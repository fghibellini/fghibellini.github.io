Some math:

\[ \begin{aligned} allRectangles = & \{ (x,y) : x \leq y \land x, y \in \{ 1 \dots n \} \} \\ area(r) = & \alpha \cdot (y-x+1) : r = (x,y), \alpha = \min_{i \in x \dots y}(h_i) \\ \mathcal{H} = & \{ r : area(r) = \max_{s \in allRectangles}{area(s)} \} \end{aligned} \]

Some code:

h = [3, 1, 1, 5, 7, 9]
n = length h

allRectangles = [ (x,y) | x <- [1..n], y <- [1..n], x < y ]
allAreas = [ getArea r | r <- allRectangles ]
maxArea = maximum allAreas
results = [ r | r <- allRectangles, getArea r == maxArea ]

getArea :: (Int, Int) -> Int
getArea (start, end) = let width = end - start + 1
                           subhistogram = (take width (drop start h))
                           height = minimum subhistogram
                       in height * width
                       
print results
